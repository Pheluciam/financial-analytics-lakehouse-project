# Loads .env into os.environ — secrets externalised per criterion 4 of

python-dotenv>=1.0.0

# HTTP client for SEC EDGAR — added session 2 ahead of extract_sec_edgar.py.

# dbt adapter for Amazon Athena — Phase 2 Silver Data Vault 2.0 modeling.
# Pulls dbt-core, dbt-adapters, pyathena (the Athena driver), and the AWS
# SDK stubs as transitive dependencies. dbt-athena-community is the
# backwards-compat distribution name; the underlying package is dbt-athena,
# now maintained by dbt Labs in the dbt-labs/dbt-adapters monorepo.
dbt-athena-community>=1.10.1

# dbt-core upper bound — Phase 3 orchestration constraint. dbt-core 1.11+
# requires Python >=3.10, but Glue Python Shell maxes out at Python 3.9
# (Risk 26). Pinning dbt-core to the 1.10.x line keeps the local venv
# byte-aligned with the Glue runtime per Risk 25's stability contract.
# Banked as Risk 30 (Phase 3 session 12) — managed-runtime Python ceiling
# vs tool Python floor cross-check at every adapter-or-tool version bump.
dbt-core>=1.10.0,<1.11

# statsmodels — classical time-series stack for Phase 4 session 4 forecast
# pipeline (scripts/forecast.py). Provides statsmodels.tsa.holtwinters.ExponentialSmoothing
# (per-company annual-revenue Holt-Winters fit + prediction intervals) and
# statsmodels.tsa.arima.ARIMA (fallback for non-trended series). Pure-pip
# install with NumPy + SciPy + Pandas + Patsy transitive deps; no C++
# compilation step (vs Prophet's Stan backend). Risk 38 lock (Phase 3
# session 14 forward-verify) — chosen on annual-cadence financial time
# series workload: 10 fiscal year-ends × 100 companies × ~1 in-scope
# canonical with 3-year forecast horizon + 95% prediction intervals.
statsmodels>=0.14

# pandas — DataFrame + groupby surface for the per-company forecast loop
# in scripts/forecast.py. Already a transitive dep via dbt-core + statsmodels;
# pinned explicitly here so the forecast script's import is self-documenting
# in requirements.txt.
pandas>=2.0

# pyarrow — Parquet writer for the forecast pipeline output landing under
# s3://phil-financial-analytics-lakehouse/forecasts/. Backs pandas.to_parquet
# with the 'pyarrow' engine. Transitive via pyathena but pinned explicitly
# here so the Parquet write path is self-documenting in requirements.txt.
pyarrow>=14.0
